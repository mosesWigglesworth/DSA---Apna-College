```java
import java.util.Scanner;

public class SumNaturalNumbers {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int value = 1;
        int sum = 0;

        while (value <= n) {
            sum += value;
            value++;
        }

        System.out.println(sum);
    }
}
```
